-- GoldCraft Advisor - Calculator.lua
-- ROI and Profit calculation engine

local _, GCA = ...

-- ================= Constants =================
local AH_CUT = 0.05  -- 5% auction house fee
local AH_MULTIPLIER = 1 - AH_CUT  -- 0.95

-- ================= Data Completeness Check =================
-- CRITICAL: Never calculate ROI with missing prices

function GCA:IsStrategyComplete(strategyName)
    local strategy = self.Strategies[strategyName]
    if not strategy then
        return false, "Unknown strategy"
    end

    -- Check all input prices
    for _, input in ipairs(strategy.inputs) do
        local price = self:GetPrice(input.itemID, input.rank)
        if not price or price <= 0 then
            local itemName = self:GetItemName(input.itemID)
            return false, "Missing input price: " .. (input.label or itemName or input.itemID)
        end
    end

    -- Check all output prices
    for _, output in ipairs(strategy.outputs) do
        local price = self:GetPrice(output.itemID, output.rank)
        if not price or price <= 0 then
            local itemName = self:GetItemName(output.itemID)
            return false, "Missing output price: " .. (output.label or itemName or output.itemID)
        end
    end

    return true
end

-- ================= Calculate Strategy ROI =================

function GCA:CalculateStrategyROI(strategyName)
    local strategy = self.Strategies[strategyName]
    if not strategy then
        return nil, "Unknown strategy: " .. tostring(strategyName)
    end

    -- CRITICAL: Check data completeness first
    local complete, missing = self:IsStrategyComplete(strategyName)
    if not complete then
        return nil, missing
    end

    -- Calculate total input cost
    local totalCost = 0
    for _, input in ipairs(strategy.inputs) do
        local price = self:GetPrice(input.itemID, input.rank)
        local quantity = input.quantity or 1
        totalCost = totalCost + (price * quantity)
    end

    if totalCost <= 0 then
        return nil, "Total cost is zero or negative"
    end

    -- Calculate total output value
    local totalValue = 0
    local baseInputQuantity = strategy.inputs[1].quantity or 1

    for _, output in ipairs(strategy.outputs) do
        local price = self:GetPrice(output.itemID, output.rank)
        local avgPer = output.avgPer or 1

        -- Calculate expected quantity
        local expectedQty = avgPer * baseInputQuantity

        -- Apply resourcefulness bonus if applicable
        local resourcefulness = strategy.resourcefulness or 0
        if resourcefulness > 0 then
            expectedQty = expectedQty * (1 + resourcefulness)
        end

        -- Apply multicraft bonus if applicable
        local multicraft = strategy.multicraft or 0
        if multicraft > 0 then
            expectedQty = expectedQty * (1 + multicraft)
        end

        totalValue = totalValue + (price * expectedQty)
    end

    -- Calculate profit after AH cut
    local revenue = totalValue * AH_MULTIPLIER
    local profit = revenue - totalCost
    local roi = (profit / totalCost) * 100

    return {
        strategyName = strategyName,
        category = strategy.category,
        name = strategy.name,
        cost = totalCost,
        value = totalValue,
        revenue = revenue,
        profit = profit,
        roi = roi,
        complete = true,
    }
end

-- ================= Calculate All Strategies =================

function GCA:CalculateAllROI()
    local results = {}

    for strategyName, strategy in pairs(self.Strategies) do
        local result, errorMsg = self:CalculateStrategyROI(strategyName)

        if result then
            results[#results + 1] = result
        else
            -- Include incomplete strategies with error message
            results[#results + 1] = {
                strategyName = strategyName,
                category = strategy.category,
                name = strategy.name,
                complete = false,
                error = errorMsg,
            }
        end
    end

    -- Sort by ROI (highest first), incomplete at the end
    table.sort(results, function(a, b)
        if a.complete and b.complete then
            return (a.roi or 0) > (b.roi or 0)
        elseif a.complete then
            return true
        elseif b.complete then
            return false
        else
            return a.strategyName < b.strategyName
        end
    end)

    -- Store results
    self.lastResults = results

    -- Print summary
    local completeCount = 0
    local profitableCount = 0
    local bestROI = nil
    local bestStrategy = nil

    for _, result in ipairs(results) do
        if result.complete then
            completeCount = completeCount + 1
            if result.roi and result.roi > 0 then
                profitableCount = profitableCount + 1
                if not bestROI or result.roi > bestROI then
                    bestROI = result.roi
                    bestStrategy = result.name
                end
            end
        end
    end

    print(string.format("|cff00ff00[GCA]|r Calculated %d strategies (%d complete, %d profitable)",
        #results, completeCount, profitableCount))

    if bestStrategy and bestROI then
        print(string.format("|cff00ff00[GCA]|r Best ROI: %s (%.1f%%)", bestStrategy, bestROI))
    end

    -- Update UI if available
    if self.UpdateResults then
        self:UpdateResults()
    end

    return results
end

-- ================= Get Results by Category =================

function GCA:GetResultsByCategory(category)
    if not self.lastResults then
        self:CalculateAllROI()
    end

    if not category or category == "All" then
        return self.lastResults
    end

    local filtered = {}
    for _, result in ipairs(self.lastResults) do
        if result.category == category then
            filtered[#filtered + 1] = result
        end
    end

    return filtered
end

-- ================= Get Best Strategies =================

function GCA:GetBestStrategies(limit)
    limit = limit or 10

    if not self.lastResults then
        self:CalculateAllROI()
    end

    local best = {}
    for i, result in ipairs(self.lastResults) do
        if result.complete and result.roi and result.roi > 0 then
            best[#best + 1] = result
            if #best >= limit then
                break
            end
        end
    end

    return best
end

-- ================= Format Results =================

function GCA:FormatResult(result)
    if not result then return "No data" end

    if not result.complete then
        return string.format("%s: Incomplete (%s)", result.name, result.error or "missing data")
    end

    return string.format("%s: Cost=%.0fg Value=%.0fg Profit=%.0fg ROI=%.1f%%",
        result.name,
        result.cost / 10000,
        result.value / 10000,
        result.profit / 10000,
        result.roi)
end

-- ================= Print All Results =================

function GCA:PrintAllResults()
    if not self.lastResults then
        self:CalculateAllROI()
    end

    print("|cff00ff00[GCA]|r === Strategy Results ===")

    for i, result in ipairs(self.lastResults) do
        if i <= 20 then  -- Limit output
            local color = result.complete and (result.roi > 0 and "|cff00ff00" or "|cffff0000") or "|cff888888"
            print(color .. self:FormatResult(result) .. "|r")
        end
    end

    if #self.lastResults > 20 then
        print(string.format("|cff888888... and %d more strategies|r", #self.lastResults - 20))
    end
end

-- ================= Detailed Breakdown =================

function GCA:GetDetailedBreakdown(strategyName)
    local strategy = self.Strategies[strategyName]
    if not strategy then
        return nil, "Unknown strategy"
    end

    local breakdown = {
        strategyName = strategyName,
        category = strategy.category,
        name = strategy.name,
        guide = strategy.guide,
        inputs = {},
        outputs = {},
        totals = {},
    }

    -- Input details
    local totalCost = 0
    for _, input in ipairs(strategy.inputs) do
        local price = self:GetPrice(input.itemID, input.rank)
        local quantity = input.quantity or 1
        local cost = price and (price * quantity) or nil

        breakdown.inputs[#breakdown.inputs + 1] = {
            itemID = input.itemID,
            rank = input.rank,
            quantity = quantity,
            label = input.label or self:GetItemName(input.itemID),
            unitPrice = price,
            totalCost = cost,
            hasPrice = price and price > 0,
        }

        if cost then
            totalCost = totalCost + cost
        end
    end

    -- Output details
    local totalValue = 0
    local baseInputQuantity = strategy.inputs[1].quantity or 1

    for _, output in ipairs(strategy.outputs) do
        local price = self:GetPrice(output.itemID, output.rank)
        local avgPer = output.avgPer or 1
        local expectedQty = avgPer * baseInputQuantity

        -- Apply bonuses
        local resourcefulness = strategy.resourcefulness or 0
        local multicraft = strategy.multicraft or 0
        if resourcefulness > 0 then
            expectedQty = expectedQty * (1 + resourcefulness)
        end
        if multicraft > 0 then
            expectedQty = expectedQty * (1 + multicraft)
        end

        local value = price and (price * expectedQty) or nil

        breakdown.outputs[#breakdown.outputs + 1] = {
            itemID = output.itemID,
            rank = output.rank,
            avgPer = avgPer,
            expectedQuantity = expectedQty,
            label = output.label or self:GetItemName(output.itemID),
            unitPrice = price,
            totalValue = value,
            hasPrice = price and price > 0,
        }

        if value then
            totalValue = totalValue + value
        end
    end

    -- Totals
    local allComplete = true
    for _, input in ipairs(breakdown.inputs) do
        if not input.hasPrice then allComplete = false break end
    end
    for _, output in ipairs(breakdown.outputs) do
        if not output.hasPrice then allComplete = false break end
    end

    if allComplete and totalCost > 0 then
        local revenue = totalValue * AH_MULTIPLIER
        local profit = revenue - totalCost
        local roi = (profit / totalCost) * 100

        breakdown.totals = {
            cost = totalCost,
            value = totalValue,
            revenue = revenue,
            profit = profit,
            roi = roi,
            complete = true,
        }
    else
        breakdown.totals = {
            complete = false,
            cost = totalCost,
            value = totalValue,
        }
    end

    breakdown.resourcefulness = strategy.resourcefulness
    breakdown.multicraft = strategy.multicraft

    return breakdown
end

-- ================= Print Detailed Breakdown =================

function GCA:PrintDetailedBreakdown(strategyName)
    local breakdown = self:GetDetailedBreakdown(strategyName)
    if not breakdown then
        print("|cffff0000[GCA]|r Unknown strategy: " .. tostring(strategyName))
        return
    end

    print("|cff00ff00[GCA]|r === " .. breakdown.name .. " ===")
    print("Category: " .. breakdown.category)
    if breakdown.guide then
        print("Setup: " .. breakdown.guide)
    end

    print("\nInputs:")
    for _, input in ipairs(breakdown.inputs) do
        local priceStr = input.hasPrice and string.format("%.2fg", input.unitPrice / 10000) or "MISSING"
        local costStr = input.totalCost and string.format("%.0fg", input.totalCost / 10000) or "---"
        print(string.format("  %s x%d @ %s = %s", input.label, input.quantity, priceStr, costStr))
    end

    print("\nOutputs:")
    for _, output in ipairs(breakdown.outputs) do
        local priceStr = output.hasPrice and string.format("%.2fg", output.unitPrice / 10000) or "MISSING"
        local valueStr = output.totalValue and string.format("%.0fg", output.totalValue / 10000) or "---"
        print(string.format("  %s x%.1f @ %s = %s",
            output.label, output.expectedQuantity, priceStr, valueStr))
    end

    if breakdown.resourcefulness and breakdown.resourcefulness > 0 then
        print(string.format("\nResourcefulness: %.1f%%", breakdown.resourcefulness * 100))
    end
    if breakdown.multicraft and breakdown.multicraft > 0 then
        print(string.format("Multicraft: %.1f%%", breakdown.multicraft * 100))
    end

    print("\nTotals:")
    if breakdown.totals.complete then
        print(string.format("  Cost: %.0fg", breakdown.totals.cost / 10000))
        print(string.format("  Value: %.0fg", breakdown.totals.value / 10000))
        print(string.format("  Revenue (after 5%% AH): %.0fg", breakdown.totals.revenue / 10000))
        print(string.format("  Profit: %.0fg", breakdown.totals.profit / 10000))
        print(string.format("  ROI: %.1f%%", breakdown.totals.roi))
    else
        print("  INCOMPLETE - Missing price data")
    end
end
